  for each vertex V in G
    distance[V] <- infinite
      previous[V] <- NULL
  distance[S] <- 0\n
  for each vertex V in G
    for each edge (U,V) in G
      tempDistance <- distance[U] + edge_weight(U, V)
      if tempDistance < distance[V]
        distance[V] <- tempDistance
        previous[V] <- U\n
