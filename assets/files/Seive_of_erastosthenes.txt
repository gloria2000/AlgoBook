void SieveOfEratosthenes(int n) 
{ 
    // Create a boolean array prime[0..n] and initialize 
    // all entries it as true. A value in prime[i] will 
    // finally be false if i is Not a prime, else true. 
    bool prime[n+1]; 
    memset(prime, true, sizeof(prime));

    {
